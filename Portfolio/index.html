ARITHMETIC

This takes your inputs and calculates the smallest, largest, average, sum, and product:

                                    function small(array) {
                                        var x = array[0];
                                        for (var y of array) {
                                            if (y < x) {
                                                x = y;
                                            };
                                        };
                                        return x;
                                    };
                                    function big(array) {
                                        var x = array[0];
                                        for (var y of array) {
                                            if (y > x) {
                                                x = y;
                                            };
                                        };
                                        return x;
                                    };
                                    function avg(array) {
                                        var count = array.length;
                                        var add = sum(array);
                                        return add / count;
                                    };
                                    function sum(array) {
                                        var x = 0;
                                        for (var i of array) {
                                            x += i;
                                        };
                                        return x;
                                    };
                                    function prod(array) {
                                        var x = 1;
                                        for (var i of array) {
                                            x *= i;
                                        };
                                        return x;
                                    };

Factorial

This will test your input if it is the product of an integer and all the integers below it:

                                    $("#doMathFact").click(function () {
                                        var factNum = Number($("#inputFact").val());

                                        function fact(x) {
                                            if (x === 1 || x === 0) {
                                                return 1;
                                            } for (var i = 2; i <= x - 1; i++) {
                                                factNum *= i;
                                            };
                                            return factNum;
                                        };

                                        $("#factOutput").text(`The factorial of ${factNum} is ${fact(factNum)}.`);
                                    });

FizzBuzz

This finds the multiples of your inputs and labels them, "Fizz" or "Buzz" accordingly. Any shared multiples will be labled, "FizzBuzz":

                                    $("#pop").click(function () {
                                        var fb1 = Number($("#inputFB1").val());
                                        var fb2 = Number($("#inputFB2").val());
                                        var arrayFB = [];

                                        function fizzbuzz(x, y) {
                                            for (var i = 1; i <= 100; i++) {
                                                if (i % x == 0 && i % y == 0) {
                                                    arrayFB.push("FizzBuzz");
                                                } else if (i % x == 0) {
                                                    arrayFB.push("Fizz");
                                                } else if (i % y == 0) {
                                                    arrayFB.push("Buzz");
                                                } else {
                                                    arrayFB.push(i);
                                                };
                                            };
                                            return arrayFB.join(", ")
                                        };

                                        $("#outputFB").text(`${fizzbuzz(fb1, fb2)}`);
                                    });

Palindrome

This will test your input if it is a word, phrase, or sequence that reads the same backward as forward:

                                        $("#reverse").click(function () {
                                            var word1 = $("#inputPD").attr("checked", true).val();
                                            var word2 = "";

                                            function reverse(x) {
                                                for (var i = x.length - 1; i >= 0; i--) {
                                                    word2 += x[i];
                                                };
                                                return word2;
                                            };

                                            reverse(word1)
                                            var pword1 = word1.toLowerCase().replace(/[^\w]|_/g, "");
                                            var pword2 = word2.toLowerCase().replace(/[^\w]|_/g, "");
                                            pword1 == pword2 ? $("#outputPD").text(`SUCCESS! ${word1} reversed is: ${word2}`) : $("#outputPD").text(`UH-OH! ${word1} reversed is: ${word2}`);
                                        });
